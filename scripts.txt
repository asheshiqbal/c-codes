// Palindrome Check
#include <stdio.h>
int main()
{
    int i,l, flag=1;
    char s[100];
    gets(s);
    l = strlen(s);
    for(i=0;i<(l/2);i++)
    {
        if((s[i])!=(s[l-(i+1)]))
        {
            flag=0;
            break;
        }
    }
    if(flag==1)
        printf("Y");
    else printf("N");
}


// Fibonacci
#include<stdio.h>

int main()
{
    int t1=0, t2=1, i, next, n=10;
    
    for (i = 0; i < n; i++)
    {
        printf("%d ", t1);
        next = t1 + t2;
        t1 = t2;
        t2 = next;
    }
}

// Convergance by Summation
#include<stdio.h>

int main()
{
    int s=0,n=1234;
    while(n>0)
    {
        while(n>0)
        {
            s+=n%10;
            n=n/10;
        }
        printf("%d ",s);
        if(s>9)
        {
            n=s;
            s=0;
        }   
    }
}

// Max Min
#include<stdio.h>

int main()
{
    int ip,i,n,max=-100000,min=100000;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&ip);
        if(ip>max)
        max=ip;
        if(ip<min)
        min=ip;
    }
    printf("%d %d",max,min);
}

// Factorial
#include<stdio.h>

int f(int a)
{
    if(a>1)
    return(a*f(a-1));
    else return(1);
}
